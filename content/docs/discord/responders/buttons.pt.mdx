Responda um bot찾o definindo o tipo do `Responder` para bot찾o.

```ts title="command.ts"
const row = createRow(
    new ButtonBuilder({
        customId: "confirm/button", 
        label: "Confirmar", 
        style: ButtonStyle.Success
    })
);
interaction.reply({ ephemeral, components: [row] });
```

```ts title="responder.ts"
new Responder({
    customId: "confirm/button",
    type: ResponderType.Button, cache: "cached",
    async run(interaction) {
        interaction.update({ 
            ephemeral, content: "Confirmado", 
            components: [] 
        });
    },
});
```
